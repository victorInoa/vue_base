src/
├── api/             # Centralized Axios configurations and base API clients
├── assets/          # Static files like images, fonts, global CSS/SCSS
├── components/      # General, reusable UI components (e.g., BaseButton, AppModal)
├── composables/     # Reusable logic using the Composition API (e.g., useFetch, useForm)
├── domains/         # OR features/ - Application business logic grouped by domain
│   ├── UserManagement/
│   │   ├── api/     # API calls specific to this domain
│   │   ├── components/ # Components specific to this domain
│   │   ├── composables/ # Composables specific to this domain
│   │   ├── models/  # TypeScript interfaces/types for this domain
│   │   ├── pages/   # Top-level pages/views for this domain (e.g., UsersIndex.vue)
│   │   └── store/   # Pinia stores specific to this domain
├── layouts/         # Layout components (e.g., DefaultLayout, AuthLayout)
├── router/          # Vue Router configuration, potentially split by domain
├── stores/          # Global Pinia stores (minimal, only app-wide state)
├── types/           # Global TypeScript interfaces and type definitions
├── utils/           # Generic, non-Vue specific utility functions (e.g., formatters, validators)
├── App.vue          # Root component
└── main.ts          # Application entry point, mounting logic, global registrations
